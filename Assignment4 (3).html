# Selecting the features for clustering
features_scaled = df[['Annual Income (k$)', 'Spending Score (1-100)']]

# Lists to store the metrics
wss = []  # Within-Cluster-Sum of Squared Errors for Elbow Method
silhouette_scores = []  # Silhouette Scores
range_n_clusters = range(2, 11)

# Calculate metrics for each number of clusters
for n_clusters in range_n_clusters:
    kmeans = KMeans(n_clusters=n_clusters, random_state=42)
    kmeans.fit(features_scaled)
    
    # Elbow method metric (WSS)
    wss.append(kmeans.inertia_)
    
    # Silhouette Score
    labels = kmeans.labels_
    silhouette_scores.append(silhouette_score(features_scaled, labels))

# Using Kneedle to find the elbow point
kneedle = KneeLocator(range_n_clusters, wss, curve='convex', direction='decreasing')

# Elbow point detected by the Kneedle algorithm
optimal_k_elbow = kneedle.elbow

# Plotting the Elbow Method and Silhouette Method
plt.figure(figsize=(14, 6))

# Elbow Method plot
plt.subplot(1, 2, 1)
plt.plot(range_n_clusters, wss, marker='o')
plt.axvline(optimal_k_elbow, linestyle='--', color='red', label=f'Optimal k: {optimal_k_elbow}')
plt.title('Elbow Method for Optimal k')
plt.xlabel('Number of Clusters')
plt.ylabel('WSS (Within-Cluster Sum of Squares)')
plt.legend()
plt.grid()

# Silhouette Method plot
plt.subplot(1, 2, 2)
plt.plot(range_n_clusters, silhouette_scores, marker='o')
plt.title('Silhouette Method for Optimal k')
plt.xlabel('Number of Clusters')
plt.ylabel('Silhouette Score')
plt.grid()

plt.tight_layout()
plt.show()

# Silhouette Method: Optimal k is the one with the highest silhouette score.
optimal_k_silhouette = range_n_clusters[silhouette_scores.index(max(silhouette_scores))]

# Calculate average k
best_k = (optimal_k_elbow + optimal_k_silhouette) / 2

# Round the average k to the nearest integer
best_k = round(best_k)

print(f"Optimal k based on Elbow Method: {optimal_k_elbow}")
print(f"Optimal k based on Silhouette Method: {optimal_k_silhouette}")
print(f"Final chosen best k value: {best_k} (Averaged from both methods)")

# Get the labels and centroids
labels = kmeans.labels_
centroids = kmeans.cluster_centers_

# 1. Calculate Within-Cluster Distance
within_cluster_distances = []
for cluster in range(best_k):
    cluster_points = features_scaled[labels == cluster]
    centroid = centroids[cluster]
    # Calculate distances from each point to the centroid
    distances = np.linalg.norm(cluster_points - centroid, axis=1)
    # Average distance for this cluster
    within_cluster_distances.append(np.mean(distances))

# Total Within-Cluster Distance
total_within_cluster_distance = np.mean(within_cluster_distances)
print(f"Within-Cluster Distances (per cluster): {within_cluster_distances}")
